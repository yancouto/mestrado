			\State $p_2 =$ \func{Push}{p_1, 7}
			\State $p_3 =$ \func{Push}{p_2, 6}
			\State $p_4 =$ \func{Pop}{p_2}
			\State \func{Top}{p_3}
			\State $p_5 =$ \func{Push}{p_4, 9}

			\State $p_6 =$ \func{Push}{p_0, 5}
		\end{algorithmic}
	\end{subalgorithm}
